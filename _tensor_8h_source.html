975void Tensor<T>::fill_data_shape(const std::initializer_list<std::initializer_list<T>> &list,
976 std::vector<T> &flat_data, std::vector<size_t> &shape) {
977
978 shape.emplace_back(list.size());
979 std::vector<std::initializer_list<T>> v_list(list);
980 for (size_t i{1}; i < v_list.size(); ++i)
981 if (v_list[i].size() != v_list[i - 1].size())
982 throw txeo::TensorError("Tensor initialization is inconsistent!");
983
984 shape.emplace_back(v_list[0].size());
985 for (auto &item : v_list)
986 for (auto &subitem : item)

1000 shape.emplace_back(v_list[0].size());

1003 std::vector<std::initializer_list<T>> v_sublist(v_list[i]);
